# (required for GCC and Clang, comment out in MSVC)

LIBS += -L/data/boost_1_63_0/stage/lib

# Compiler to use (comment out to use default one)
#QMAKE_CXX = clang++

# Boost specific defines
DEFINES += BOOST_THREAD_VERSION=4
